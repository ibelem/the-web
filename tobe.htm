function BuyBeijingRooms() {
var sName;
var isBeijingJitiHukou;
var isPayTaxesMoreThan5Years;

this.__defineGetter__("sName",function() { return sName});

this.__defineSetter__("sName",function(val) { sName = "你的名字: " + val});

this.__defineGetter__("isBeijingJitiHukou",function() { return isBeijingJitiHukou});

this.__defineSetter__("isBeijingJitiHukou",function(boolVal) { isBeijingJitiHukou = boolVal});

this.__defineGetter__("isPayTaxesMoreThan5Years",function() { return isPayTaxesMoreThan5Years});

this.__defineSetter__("isPayTaxesMoreThan5Years",function(boolVal) { isPayTaxesMoreThan5Years = boolVal});

this.toBeOrNotToBe = function() {
if(isBeijingJitiHukou) {
alert(sName+ "\r\n" + "户口: 北京集体户口 \r\n" + "能否买房: 非在校生可以购房, 管你丫是否交满5年社保个税!");
} else {
if(isPayTaxesMoreThan5Years) {
alert(sName+ "\r\n" + "户口: 外地 \r\n" + "能否买房: 你社保个税交齐5年, 恭喜你, 现在即可买房!");
} else {
alert(sName+ "\r\n" + "户口: 外地 \r\n" + "能否买房: 外地户口找中介补交5年社保个税即可买房.");
}
}
}

function addEventHandler(oNode, sEvt, sFunc, bCaptures) {

oNode.attachEvent("on" + sEvt, sFunc);

}
